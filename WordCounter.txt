import java.swing.*; 
import java.awt.*;
import java.io.*;
import java.util.*;

public class WordCounter {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Word Counter");
        frame.setSize(500, 400);
        frame.setLayout(new BorderLayout());

        JLabel label = new JLabel("Enter the path of your text file:");
        JTextField inputBox = new JTextField(30);
        JTextArea textArea = new JTextArea();
        textArea.setEditable(false);

        JPanel topPanel = new JPanel();
        topPanel.add(label);
        topPanel.add(inputBox);

        frame.add(topPanel, BorderLayout.NORTH);
        frame.add(new JScrollPane(textArea), BorderLayout.CENTER);

        inputBox.addActionListener(e -> {
            String filePath = inputBox.getText().trim();
            if (!filePath.isEmpty()) {
                File file = new File(filePath);
                if (file.exists() && file.isFile()) {
                    try (BufferedReader reader = new BufferedReader(new InputStreamReader(new FileInputStream(file)))) {
                        // Read and process the file
                        String line;
                        Map<String, Integer> wordFreq = new TreeMap<>();
                        while ((line = reader.readLine()) != null) {
                            String[] words = line.toLowerCase().replaceAll("[^a-zA-Z\\s]", "").split("\\s+");
                            for (String word : words) {
                                if (!word.isEmpty()) {
                                    wordFreq.put(word, wordFreq.getOrDefault(word, 0) + 1);
                                }
                            }
                        }
                        // Display the frequency report
                        StringBuilder report = new StringBuilder("Frequency Report (Ascending Order):\n");
                        wordFreq.forEach((word, freq) -> report.append(word).append(": ").append(freq).append("\n"));
                        textArea.setText(report.toString());

                    } catch (IOException ex) {
                        textArea.setText("Error reading file: " + ex.getMessage());
                    }
                } else {
                    textArea.setText("Invalid file path! Please try again.");
                }
            } else {
                textArea.setText("File path cannot be empty.");
            }
        });
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setVisible(true);
    }
}
